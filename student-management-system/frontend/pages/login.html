from pyodide.http import pyfetch import json class LoginComponent: def __init__(self): self.api_url = "http://localhost:8000" self.setup_event_listeners() print("✓ Login component initialized") def setup_event_listeners(self): """Setup all event listeners""" login_form = document.getElementById("login-form") login_form.addEventListener("submit", lambda e: self.handle_login(e)) def clear_messages(self): """Clear all messages""" error_msg = document.getElementById("error-message") success_msg = document.getElementById("success-message") error_msg.classList.remove("show") success_msg.classList.remove("show") async def handle_login(self, event): """Handle login form submission""" event.preventDefault() username = document.getElementById("login-username").value password = document.getElementById("login-password").value if not username or not password: self.show_error("Please fill in all fields") return loading = document.getElementById("login-loading") loading.classList.add("show") try: response = await pyfetch( f"{self.api_url}/auth/login", method="POST", headers={"Content-Type": "application/json"}, body=json.dumps({ "username": username, "password": password }) ) data = await response.json() if response.status == 200: self.show_success(f"Welcome back, {username}!") # Store token and redirect after 1 second document.localStorage.setItem("token", data.get("access_token")) # Uncomment to redirect: setTimeout(() => window.location.href = "/dashboard", 1000) else: self.show_error(data.get("detail", "Login failed")) except Exception as e: self.show_error(f"Connection error: {str(e)}") finally: loading.classList.remove("show") def show_error(self, message): """Display error message""" error_msg = document.getElementById("error-message") error_msg.textContent = f"✗ {message}" error_msg.classList.add("show") self.auto_hide_message(error_msg, 5000) def show_success(self, message): """Display success message""" success_msg = document.getElementById("success-message") success_msg.textContent = f"✓ {message}" success_msg.classList.add("show") self.auto_hide_message(success_msg, 4000) def auto_hide_message(self, element, duration): """Auto-hide message after duration""" def hide(): element.classList.remove("show") # Use JS timeout instead of Python from js import setTimeout setTimeout(hide, duration) # Initialize the component login_component = LoginComponent()
