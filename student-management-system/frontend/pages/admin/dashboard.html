class AdminDashboard: def __init__(self): self.check_auth() self.setup_data() self.calculate_metrics() self.setup_event_listeners() print("✓ Admin Dashboard initialized") def check_auth(self): """Check if user is authenticated and is admin""" token = document.localStorage.getItem("token") role = document.localStorage.getItem("role") if not token or role != "admin": window.location.href = "../login.html" def setup_data(self): """Setup sample data""" # Sample student data self.students = [ {"id": 1, "name": "John Smith", "email": "john@student.edu", "department": "Computer Science"}, {"id": 2, "name": "Sarah Johnson", "email": "sarah@student.edu", "department": "Computer Science"}, {"id": 3, "name": "Michael Brown", "email": "michael@student.edu", "department": "Engineering"}, {"id": 4, "name": "Emily Davis", "email": "emily@student.edu", "department": "Computer Science"}, {"id": 5, "name": "David Wilson", "email": "david@student.edu", "department": "Business"}, ] # Sample course data self.courses = [ {"id": 1, "code": "CS101", "name": "Python Basics", "instructor": "Dr. Smith"}, {"id": 2, "code": "CS201", "name": "Data Structures", "instructor": "Dr. Johnson"}, {"id": 3, "code": "ENG101", "name": "Engineering Physics", "instructor": "Prof. Williams"}, {"id": 4, "code": "BUS101", "name": "Business Fundamentals", "instructor": "Prof. Brown"}, ] # Sample grade data self.grades = [ {"id": 1, "student_id": 1, "course_id": 1, "grade": "A", "percentage": 92}, {"id": 2, "student_id": 2, "course_id": 1, "grade": "A", "percentage": 95}, {"id": 3, "student_id": 1, "course_id": 2, "grade": "B", "percentage": 85}, ] # Sample assignment data self.assignments = [ {"id": 1, "title": "Python Basics", "status": "Graded"}, {"id": 2, "title": "List Operations", "status": "Submitted"}, {"id": 3, "title": "Data Structures Project", "status": "Pending"}, ] # Sample attendance data self.attendance = [ {"id": 1, "status": "Present"}, {"id": 2, "status": "Present"}, {"id": 3, "status": "Absent"}, {"id": 4, "status": "Late"}, {"id": 5, "status": "Present"}, ] def calculate_metrics(self): """Calculate and display dashboard metrics""" # Total Students total_students = len(self.students) document.getElementById("total-students").textContent = str(total_students) # Total Courses total_courses = len(self.courses) document.getElementById("total-courses").textContent = str(total_courses) # Pending Assignments pending = sum(1 for a in self.assignments if a["status"] == "Pending") document.getElementById("pending-assignments").textContent = str(pending) # Attendance Rate if self.attendance: present_count = sum(1 for a in self.attendance if a["status"] == "Present") attendance_rate = (present_count / len(self.attendance)) * 100 else: attendance_rate = 0 document.getElementById("attendance-rate").textContent = f"{attendance_rate:.0f}%" # Average GPA if self.grades: grade_values = {"A": 4.0, "B": 3.0, "C": 2.0, "D": 1.0, "F": 0.0} total_gpa = sum(grade_values.get(g["grade"], 0) for g in self.grades) avg_gpa = total_gpa / len(self.grades) else: avg_gpa = 0 document.getElementById("avg-gpa").textContent = f"{avg_gpa:.2f}" # Total Assignments total_assignments = len(self.assignments) document.getElementById("total-assignments").textContent = str(total_assignments) # Submitted Assignments submitted = sum(1 for a in self.assignments if a["status"] != "Pending") document.getElementById("submitted").textContent = str(submitted) def setup_event_listeners(self): """Setup event listeners""" logout_btn = document.getElementById("logout-btn") logout_btn.addEventListener("click", lambda e: self.handle_logout(e)) def handle_logout(self, event): """Handle logout""" document.localStorage.removeItem("token") document.localStorage.removeItem("role") window.location.href = "../login.html" admin_dashboard = AdminDashboard()
