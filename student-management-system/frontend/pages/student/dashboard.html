class StudentDashboard: def __init__(self): self.check_auth() self.setup_data() self.render_dashboard() self.setup_event_listeners() print("✓ Student Dashboard initialized") def check_auth(self): token = document.localStorage.getItem("token") role = document.localStorage.getItem("role") if not token or role != "student": window.location.href = "../login.html" def setup_data(self): self.student_name = "John Smith" self.courses = [ {"id": 1, "name": "Python Basics", "instructor": "Dr. Smith", "progress": 85}, {"id": 2, "name": "Data Structures", "instructor": "Dr. Johnson", "progress": 72}, {"id": 3, "name": "Web Development", "instructor": "Prof. Williams", "progress": 65}, ] self.grades = [ {"id": 1, "course": "Python Basics", "grade": "A", "percentage": 92}, {"id": 2, "course": "Data Structures", "grade": "B", "percentage": 85}, ] self.assignments = [ {"id": 1, "title": "List Operations", "course": "Python Basics", "due_date": "2026-04-05", "status": "Pending"}, {"id": 2, "title": "Binary Trees", "course": "Data Structures", "due_date": "2026-04-10", "status": "Pending"}, ] self.attendance_records = [ {"status": "Present"}, {"status": "Present"}, {"status": "Late"}, {"status": "Present"}, {"status": "Present"}, ] def render_dashboard(self): document.getElementById("student-name").textContent = self.student_name # Render stats document.getElementById("enrolled-courses").textContent = str(len(self.courses)) if self.grades: grade_values = {"A": 4.0, "B": 3.0, "C": 2.0} total = sum(grade_values.get(g["grade"], 0) for g in self.grades) avg_gpa = total / len(self.grades) else: avg_gpa = 0 document.getElementById("avg-gpa").textContent = f"{avg_gpa:.2f}" present = sum(1 for a in self.attendance_records if a["status"] == "Present") attendance = (present / len(self.attendance_records)) * 100 if self.attendance_records else 0 document.getElementById("attendance-rate").textContent = f"{attendance:.0f}%" pending = sum(1 for a in self.assignments if a["status"] == "Pending") document.getElementById("pending-tasks").textContent = str(pending) # Render courses courses_html = "" for course in self.courses: courses_html += f"""
{course["name"]}
Instructor: {course["instructor"]}
Progress: {course["progress"]}%
""" document.getElementById("courses-list").innerHTML = courses_html # Render grades grades_html = "" for grade in self.grades: grades_html += f"""
{grade["course"]}
Grade: {grade["grade"]} ({grade["percentage"]}%)
""" document.getElementById("grades-list").innerHTML = grades_html # Render assignments assignments_html = "" for assign in self.assignments: assignments_html += f"""
{assign["title"]}
Course: {assign["course"]} | Due: {assign["due_date"]}
Status: {assign["status"]}
""" document.getElementById("assignments-list").innerHTML = assignments_html def setup_event_listeners(self): logout_btn = document.getElementById("logout-btn") logout_btn.addEventListener("click", lambda e: self.handle_logout(e)) def handle_logout(self, event): document.localStorage.removeItem("token") document.localStorage.removeItem("role") window.location.href = "../login.html" StudentDashboard()
