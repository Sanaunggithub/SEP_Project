class StudentCourses: def __init__(self): self.check_auth() self.setup_data() self.render_courses() self.setup_event_listeners() print("✓ Student Courses initialized") def check_auth(self): token = document.localStorage.getItem("token") role = document.localStorage.getItem("role") if not token or role != "student": window.location.href = "../login.html" def setup_data(self): self.courses = [ {"code": "CS101", "name": "Python Basics", "instructor": "Dr. Smith", "credits": 3, "status": "Active", "progress": 85}, {"code": "CS201", "name": "Data Structures", "instructor": "Dr. Johnson", "credits": 4, "status": "Active", "progress": 72}, {"code": "ENG101", "name": "Engineering Physics", "instructor": "Prof. Williams", "credits": 4, "status": "Active", "progress": 65}, {"code": "BUS101", "name": "Business Fundamentals", "instructor": "Prof. Brown", "credits": 3, "status": "Completed", "progress": 100}, ] def render_courses(self): html = "" for course in self.courses: status_color = "var(--success)" if course["status"] == "Active" else "#9ca3af" html += f"""
{course["name"]}
{course["code"]}
● {course["status"]}
Instructor: {course["instructor"]}
{course["credits"]} Credits
{course["progress"]}% Progress
Course Progress {course["progress"]}%
""" document.getElementById("courses-container").innerHTML = html def setup_event_listeners(self): logout_btn = document.getElementById("logout-btn") logout_btn.addEventListener("click", lambda e: self.handle_logout(e)) search_input = document.getElementById("search-input") search_input.addEventListener("keyup", lambda e: self.search_courses(e)) def search_courses(self, event): query = event.target.value.lower() filtered = [c for c in self.courses if query in c["name"].lower()] html = "" for course in (filtered if query else self.courses): status_color = "var(--success)" if course["status"] == "Active" else "#9ca3af" html += f"""
{course["name"]}
{course["code"]}
● {course["status"]}
Instructor: {course["instructor"]}
{course["credits"]} Credits
{course["progress"]}% Progress
Course Progress {course["progress"]}%
""" document.getElementById("courses-container").innerHTML = html def handle_logout(self, event): document.localStorage.removeItem("token") document.localStorage.removeItem("role") window.location.href = "../login.html" StudentCourses()
